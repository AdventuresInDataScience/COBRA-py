# 2. OBJECTIVE FUNCTION FOR OPTIMIZER
def backtest_strategy(params):
    """
    The optimizer calls this function with different parameter combinations.
    This uses the GLOBAL precomputed indicators - no recalculation!
    """
    # Decode parameters from optimizer
    bb_period = int(params['bb_period'])
    bb_std = params['bb_std']
    sma_period = int(params['sma_period'])
    rsi_threshold = params['rsi_threshold']
    stop_loss = params['stop_loss']
    take_profit = params['take_profit']
    
    # JUST ARRAY LOOKUPS - these are precomputed!
    bb_upper = INDICATORS['bb'].upper[(bb_period, bb_std)]
    sma_line = INDICATORS['sma'].ma[sma_period]
    rsi_values = INDICATORS['rsi'].rsi[rsi_threshold // 3 - 2]  # map to closest
    
    # Build signals from precomputed arrays (microseconds)
    entries = (
        (price_data > bb_upper) & 
        (price_data > sma_line) & 
        (rsi_values < rsi_threshold)
    )
    
    # Generate exits using stops
    exits = vbt.STOPLOSS.run(
        price_data,
        entries=entries,
        stop=[stop_loss, -take_profit],
        trailing=False
    ).exits
    
    # Run backtest
    portfolio = vbt.Portfolio.from_signals(
        price_data,
        entries=entries,
        exits=exits,
        freq='D',
        init_cash=10000,
        size=np.inf,
        max_size=1  # One position at a time
    )
    
    # Return objective (optimizer will maximize this)
    sharpe = portfolio.sharpe_ratio()
    return -sharpe if np.isnan(sharpe) else sharpe  # Handle NaN
